### 推荐的最佳实践

```python
# Step 1: 在前端检查文件大小
function validateFile(file: File) {
  if (file.size > 5_000_000) {
    // > 5MB
    alert('❌ 文件过大（> 5MB）\n' +
          '💡 改用 Assistant API 可以处理任意大小');
    return false;
  }
  
  if (file.size > 1_000_000) {
    // 1-5MB
    confirm('⚠️ 文件较大（' + (file.size / 1024 / 1024).toFixed(1) + 'MB）\n' +
            '这会消耗大量 token。确定继续？');
  }
  
  return true;
}

# Step 2: 在后端显示 token 消耗
def estimate_tokens(content_length: int) -> int:
  # 规则：1 token ≈ 4 字符
  return content_length // 4

# Step 3: 给用户清晰的反馈
def show_token_cost(tokens: int):
  if tokens < 100:
    print(f'✅ token 消耗: {tokens}（无压力）')
  elif tokens < 1000:
    print(f'⚠️ token 消耗: {tokens}（中等）')
  else:
    print(f'🔴 token 消耗: {tokens}（很高，不推荐）')
```
